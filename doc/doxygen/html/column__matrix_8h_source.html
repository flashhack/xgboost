74  inline void Init(const GHistIndexMatrix& gmat,
75  const FastHistParam& param) {
76  this->dtype = static_cast<DataType>(param.colmat_dtype);
77  /* if dtype is smaller than uint32_t, multiple bin_id's will be stored in each
78  slot of internal buffer. */
79  packing_factor_ = sizeof(uint32_t) / static_cast<size_t>(this->dtype);
80 
81  const bst_uint nfeature = static_cast<bst_uint>(gmat.cut->row_ptr.size() - 1);
82  const size_t nrow = gmat.row_ptr.size() - 1;
83 
84  // identify type of each column
85  feature_counts_.resize(nfeature);
86  type_.resize(nfeature);
87  std::fill(feature_counts_.begin(), feature_counts_.end(), 0);
88 
89  uint32_t max_val = 0;
90  XGBOOST_TYPE_SWITCH(this->dtype, {
91  max_val = static_cast<uint32_t>(std::numeric_limits<DType>::max());
92  });
93  for (bst_uint fid = 0; fid < nfeature; ++fid) {
94  CHECK_LE(gmat.cut->row_ptr[fid + 1] - gmat.cut->row_ptr[fid], max_val);
95  }
96 
97  gmat.GetFeatureCounts(&feature_counts_[0]);
98  // classify features
99  for (bst_uint fid = 0; fid < nfeature; ++fid) {
100  if (static_cast<double>(feature_counts_[fid])
101  < param.sparse_threshold * nrow) {
102  type_[fid] = kSparseColumn;
103  } else {
104  type_[fid] = kDenseColumn;
105  }
106  }
107 
108  // want to compute storage boundary for each feature
109  // using variants of prefix sum scan
110  boundary_.resize(nfeature);
111  size_t accum_index_ = 0;
112  size_t accum_row_ind_ = 0;
113  for (bst_uint fid = 0; fid < nfeature; ++fid) {
114  boundary_[fid].index_begin = accum_index_;
115  boundary_[fid].row_ind_begin = accum_row_ind_;
116  if (type_[fid] == kDenseColumn) {
117  accum_index_ += static_cast<size_t>(nrow);
118  } else {
119  accum_index_ += feature_counts_[fid];
120  accum_row_ind_ += feature_counts_[fid];
121  }
122  boundary_[fid].index_end = accum_index_;
123  boundary_[fid].row_ind_end = accum_row_ind_;
124  }
125 
126  index_.resize((boundary_[nfeature - 1].index_end
127  + (packing_factor_ - 1)) / packing_factor_);
128  row_ind_.resize(boundary_[nfeature - 1].row_ind_end);
129 
130  // store least bin id for each feature
131  index_base_.resize(nfeature);
132  for (bst_uint fid = 0; fid < nfeature; ++fid) {
133  index_base_[fid] = gmat.cut->row_ptr[fid];
134  }
135 
136  // pre-fill index_ for dense columns
137  for (bst_uint fid = 0; fid < nfeature; ++fid) {
138  if (type_[fid] == kDenseColumn) {
139  const size_t ibegin = boundary_[fid].index_begin;
140  XGBOOST_TYPE_SWITCH(this->dtype, {
141  const size_t block_offset = ibegin / packing_factor_;
142  const size_t elem_offset = ibegin % packing_factor_;
143  DType* begin = reinterpret_cast<DType*>(&index_[block_offset]) + elem_offset;
144  DType* end = begin + nrow;
145  std::fill(begin, end, std::numeric_limits<DType>::max());
146  // max() indicates missing values
147  });
148  }
149  }
150 
151  // loop over all rows and fill column entries
152  // num_nonzeros[fid] = how many nonzeros have this feature accumulated so far?
153  std::vector<size_t> num_nonzeros;
154  num_nonzeros.resize(nfeature);
155  std::fill(num_nonzeros.begin(), num_nonzeros.end(), 0);
156  for (size_t rid = 0; rid < nrow; ++rid) {
157  const size_t ibegin = gmat.row_ptr[rid];
158  const size_t iend = gmat.row_ptr[rid + 1];
159  size_t fid = 0;
160  for (size_t i = ibegin; i < iend; ++i) {
161  const uint32_t bin_id = gmat.index[i];
162  while (bin_id >= gmat.cut->row_ptr[fid + 1]) {
163  ++fid;
164  }
165  if (type_[fid] == kDenseColumn) {
166  XGBOOST_TYPE_SWITCH(this->dtype, {
167  const size_t block_offset = boundary_[fid].index_begin / packing_factor_;
168  const size_t elem_offset = boundary_[fid].index_begin % packing_factor_;
169  DType* begin = reinterpret_cast<DType*>(&index_[block_offset]) + elem_offset;
170  begin[rid] = static_cast<DType>(bin_id - index_base_[fid]);
171  });
172  } else {
173  XGBOOST_TYPE_SWITCH(this->dtype, {
174  const size_t block_offset = boundary_[fid].index_begin / packing_factor_;
175  const size_t elem_offset = boundary_[fid].index_begin % packing_factor_;
176  DType* begin = reinterpret_cast<DType*>(&index_[block_offset]) + elem_offset;
177  begin[num_nonzeros[fid]] = static_cast<DType>(bin_id - index_base_[fid]);
178  });
179  row_ind_[boundary_[fid].row_ind_begin + num_nonzeros[fid]] = rid;
180  ++num_nonzeros[fid];
181  }
182  }
183  }

199  const size_t elem_offset = boundary_[fid].index_begin % packing_factor_;
200  c.index = reinterpret_cast<const T*>(&index_[block_offset]) + elem_offset;
