241 template <typename TrainingParams, typename T>
242 XGBOOST_DEVICE inline T CalcGain(const TrainingParams &p, T sum_grad, T sum_hess) {
243  if (sum_hess < p.min_child_weight)
244  return T(0.0);
245  if (p.max_delta_step == 0.0f) {
246  if (p.reg_alpha == 0.0f) {
247  return Sqr(sum_grad) / (sum_hess + p.reg_lambda);
248  } else {
249  return Sqr(ThresholdL1(sum_grad, p.reg_alpha)) /
250  (sum_hess + p.reg_lambda);
251  }
252  } else {
253  T w = CalcWeight(p, sum_grad, sum_hess);
254  T ret = sum_grad * w + T(0.5) * (sum_hess + p.reg_lambda) * Sqr(w);
255  if (p.reg_alpha == 0.0f) {
256  return T(-2.0) * ret;
257  } else {
258  return T(-2.0) * (ret + p.reg_alpha * std::abs(w));
259  }
260  }
261 }

275 // calculate weight given the statistics
276 template <typename TrainingParams, typename T>
277 XGBOOST_DEVICE inline T CalcWeight(const TrainingParams &p, T sum_grad,
278  T sum_hess) {
279  if (sum_hess < p.min_child_weight)
280  return 0.0;
281  T dw;
282  if (p.reg_alpha == 0.0f) {
283  dw = -sum_grad / (sum_hess + p.reg_lambda);
284  } else {
285  dw = -ThresholdL1(sum_grad, p.reg_alpha) / (sum_hess + p.reg_lambda);
286  }
287  if (p.max_delta_step != 0.0f) {
288  if (dw > p.max_delta_step)
289  dw = p.max_delta_step;
290  if (dw < -p.max_delta_step)
291  dw = -p.max_delta_step;
292  }
293  return dw;
294 }
