100 inline Float8 pow2n(Float8 const& n) {
101  const float pow2_23 = 8388608.0; // 2^23
102  const float bias = 127.0; // bias in exponent
103  Float8 a =
104  n + Float8(bias + pow2_23); // put n + bias in least significant bits
105  __m256i b = _mm256_castps_si256(a.x);
106 
107  // Do bit shift in SSE so we don't have to use AVX2 instructions
108  __m128i c1 = _mm256_castsi256_si128(b);
109  b = _mm256_permute2f128_si256(b, b, 1);
110  __m128i c2 = _mm256_castsi256_si128(b);
111  c1 = _mm_slli_epi32(c1, 23);
112  c2 = _mm_slli_epi32(c2, 23);
113 
114  __m256i c = _mm256_insertf128_si256(_mm256_castsi128_si256(c1), (c2), 0x1);
115  return Float8(_mm256_castsi256_ps(c));
116 }

128 // AVX exp Function based off Agner Fog's vector library
129 // https://github.com/darealshinji/vectorclass/blob/master/vectormath_exp.h
130 // Modified so it doesn't require AVX2 instructions
131 // Clamps input values to the range -87.3f, +87.3f
132 inline Float8 ExpAgner(Float8 x) {
133  // Clamp input values
134  float max_x = 87.3f;
135  x = std::min(x, Float8(max_x));
136  x = std::max(x, Float8(-max_x));
137 
138  // 1/log(2)
139  const float log2e = 1.44269504088896340736f;
140 
141  // Taylor coefficients
142  const float P0expf = 1.f / 2.f;
143  const float P1expf = 1.f / 6.f;
144  const float P2expf = 1.f / 24.f;
145  const float P3expf = 1.f / 120.f;
146  const float P4expf = 1.f / 720.f;
147  const float P5expf = 1.f / 5040.f;
148 
149  const float ln2f_hi = 0.693359375f;
150  const float ln2f_lo = -2.12194440e-4f;
151 
152  Float8 r = round(x * Float8(log2e));
153  x -= r * Float8(ln2f_hi);
154  x -= r * Float8(ln2f_lo);
155 
156  Float8 x2 = x * x;
157  Float8 z = polynomial_5(x, P0expf, P1expf, P2expf, P3expf, P4expf, P5expf);
158  z *= x2;
159  z += x;
160 
161  // multiply by power of 2
162  Float8 n2 = pow2n(r);
163 
164  z = (z + Float8(1.0f)) * n2;
165  return z;
166 }
