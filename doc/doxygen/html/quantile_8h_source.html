347  if (src.size <= maxsize) {

349  }
350  RType begin = src.data[0].rmax;
351  // n is number of points exclude the min/max points
352  size_t n = maxsize - 2, nbig = 0;
353  // these is the range of data exclude the min/max point
354  RType range = src.data[src.size - 1].rmin - begin;
355  // prune off zero weights
356  if (range == 0.0f || maxsize <= 2) {
357  // special case, contain only two effective data pts
358  this->data[0] = src.data[0];
359  this->data[1] = src.data[src.size - 1];
360  this->size = 2;
361  return;
362  } else {
363  range = std::max(range, static_cast<RType>(1e-3f));
364  }
365  // Get a big enough chunk size, bigger than range / n
366  // (multiply by 2 is a safe factor)
367  const RType chunk = 2 * range / n;
368  // minimized range
369  RType mrange = 0;
370  {
371  // first scan, grab all the big chunk
372  // moving block index, exclude the two ends.
373  size_t bid = 0;
374  for (size_t i = 1; i < src.size - 1; ++i) {
375  // detect big chunk data point in the middle
376  // always save these data points.
377  if (CheckLarge(src.data[i], chunk)) {
378  if (bid != i - 1) {
379  // accumulate the range of the rest points
380  mrange += src.data[i].rmax_prev() - src.data[bid].rmin_next();
381  }
382  bid = i; ++nbig;
383  }
384  }
385  if (bid != src.size - 2) {
386  mrange += src.data[src.size-1].rmax_prev() - src.data[bid].rmin_next();

388  }

390  if (nbig >= n) {

399  this->size = 1;
400  // The counter on the rest of points, to be selected equally from small chunks.
401  n = n - nbig;
402  // find the rest of point
403  size_t bid = 0, k = 1, lastidx = 0;
404  for (size_t end = 1; end < src.size; ++end) {
405  if (end == src.size - 1 || CheckLarge(src.data[end], chunk)) {

407  size_t i = bid;

414  if (i == end) break;

705  if (inqueue.queue.size() == 1) {
706  inqueue.queue.resize(limit_size * 2);
707  } else {
708  temp.Reserve(limit_size * 2);
709  inqueue.MakeSummary(&temp);
710  // cleanup queue
711  inqueue.qtail = 0;
712  this->PushTemp();
713  }
714  }
715  inqueue.Push(x, w);
716  }
718  inline void PushTemp() {
719  temp.Reserve(limit_size * 2);
720  for (size_t l = 1; true; ++l) {
721  this->InitLevel(l + 1);
722  // check if level l is empty
723  if (level[l].size == 0) {
724  level[l].SetPrune(temp, limit_size);
725  break;
726  } else {
727  // level 0 is actually temp space
728  level[0].SetPrune(temp, limit_size);
729  temp.SetCombine(level[0], level[l]);
730  if (temp.size > limit_size) {
731  // try next level
732  level[l].size = 0;
733  } else {
734  // if merged record is still smaller, no need to send to next level
735  level[l].CopyFrom(temp); break;
736  }
737  }
738  }
739  }
741  inline void GetSummary(SummaryContainer *out) {
742  if (level.size() != 0) {
743  out->Reserve(limit_size * 2);
744  } else {
745  out->Reserve(inqueue.queue.size());
746  }
747  inqueue.MakeSummary(out);
748  if (level.size() != 0) {
749  level[0].SetPrune(*out, limit_size);
750  for (size_t l = 1; l < level.size(); ++l) {
751  if (level[l].size == 0) continue;
752  if (level[0].size == 0) {
753  level[0].CopyFrom(level[l]);
754  } else {
755  out->SetCombine(level[0], level[l]);
756  level[0].SetPrune(*out, limit_size);
757  }
758  }
759  out->CopyFrom(level[0]);
760  } else {
761  if (out->size > limit_size) {
762  temp.Reserve(limit_size);
763  temp.SetPrune(*out, limit_size);
764  out->CopyFrom(temp);
765  }
766  }
767  }
768  // used for debug, check if the sketch is valid
769  inline void CheckValid(RType eps) const {
770  for (size_t l = 1; l < level.size(); ++l) {
771  level[l].CheckValid(eps);
772  }
773  }
774  // initialize level space to at least nlevel
775  inline void InitLevel(size_t nlevel) {
776  if (level.size() >= nlevel) return;
777  data.resize(limit_size * nlevel);
778  level.resize(nlevel, Summary(NULL, 0));
779  for (size_t l = 0; l < level.size(); ++l) {
780  level[l].data = dmlc::BeginPtr(data) + l * limit_size;
781  }
782  }
783  // input data queue
784  typename Summary::Queue inqueue;
785  // number of levels
786  size_t nlevel;
787  // size of summary in each level
788  size_t limit_size;
789  // the level of each summaries
790  std::vector<Summary> level;
791  // content of the summary
792  std::vector<Entry> data;
793  // temporal summary, used for temp-merge
794  SummaryContainer temp;
795 };
