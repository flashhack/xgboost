641 inline void RegTree::CalculateContributionsApprox(const RegTree::FVec& feat, unsigned root_id,
642  bst_float *out_contribs) const {
643  CHECK_GT(this->node_mean_values.size(), 0U);
644  // this follows the idea of http://blog.datadive.net/interpreting-random-forests/
645  bst_float node_value;
646  unsigned split_index;
647  int pid = static_cast<int>(root_id);
648  // update bias value
649  node_value = this->node_mean_values[pid];
650  out_contribs[feat.size()] += node_value;
651  if ((*this)[pid].is_leaf()) {
652  // nothing to do anymore
653  return;
654  }
655  while (!(*this)[pid].is_leaf()) {
656  split_index = (*this)[pid].split_index();
657  pid = this->GetNext(pid, feat.fvalue(split_index), feat.is_missing(split_index));
658  bst_float new_value = this->node_mean_values[pid];
659  // update feature weight
660  out_contribs[split_index] += new_value - node_value;
661  node_value = new_value;
662  }
663  bst_float leaf_value = (*this)[pid].leaf_value();
664  // update leaf feature weight
665  out_contribs[split_index] += leaf_value - node_value;
666 }

668 // extend our decision path with a fraction of one and zero extensions
669 inline void ExtendPath(PathElement *unique_path, unsigned unique_depth,
670  bst_float zero_fraction, bst_float one_fraction, int feature_index) {
671  unique_path[unique_depth].feature_index = feature_index;
672  unique_path[unique_depth].zero_fraction = zero_fraction;
673  unique_path[unique_depth].one_fraction = one_fraction;
674  unique_path[unique_depth].pweight = (unique_depth == 0 ? 1.0f : 0.0f);
675  for (int i = unique_depth - 1; i >= 0; i--) {
676  unique_path[i+1].pweight += one_fraction * unique_path[i].pweight * (i + 1)
677  / static_cast<bst_float>(unique_depth + 1);
678  unique_path[i].pweight = zero_fraction * unique_path[i].pweight * (unique_depth - i)
679  / static_cast<bst_float>(unique_depth + 1);
680  }

683 // undo a previous extension of the decision path
684 inline void UnwindPath(PathElement *unique_path, unsigned unique_depth, unsigned path_index) {
685  const bst_float one_fraction = unique_path[path_index].one_fraction;
686  const bst_float zero_fraction = unique_path[path_index].zero_fraction;
687  bst_float next_one_portion = unique_path[unique_depth].pweight;
688 
689  for (int i = unique_depth - 1; i >= 0; --i) {
690  if (one_fraction != 0) {
691  const bst_float tmp = unique_path[i].pweight;
692  unique_path[i].pweight = next_one_portion * (unique_depth + 1)
693  / static_cast<bst_float>((i + 1) * one_fraction);
694  next_one_portion = tmp - unique_path[i].pweight * zero_fraction * (unique_depth - i)
695  / static_cast<bst_float>(unique_depth + 1);
696  } else {
697  unique_path[i].pweight = (unique_path[i].pweight * (unique_depth + 1))
698  / static_cast<bst_float>(zero_fraction * (unique_depth - i));
699  }
700  }
701 
702  for (auto i = path_index; i < unique_depth; ++i) {
703  unique_path[i].feature_index = unique_path[i+1].feature_index;
704  unique_path[i].zero_fraction = unique_path[i+1].zero_fraction;
705  unique_path[i].one_fraction = unique_path[i+1].one_fraction;
706  }
707 }

709 // determine what the total permuation weight would be if
710 // we unwound a previous extension in the decision path
711 inline bst_float UnwoundPathSum(const PathElement *unique_path, unsigned unique_depth,
712  unsigned path_index) {
713  const bst_float one_fraction = unique_path[path_index].one_fraction;
714  const bst_float zero_fraction = unique_path[path_index].zero_fraction;
715  bst_float next_one_portion = unique_path[unique_depth].pweight;
716  bst_float total = 0;
717  for (int i = unique_depth - 1; i >= 0; --i) {
718  if (one_fraction != 0) {
719  const bst_float tmp = next_one_portion * (unique_depth + 1)
720  / static_cast<bst_float>((i + 1) * one_fraction);
721  total += tmp;
722  next_one_portion = unique_path[i].pweight - tmp * zero_fraction * ((unique_depth - i)
723  / static_cast<bst_float>(unique_depth + 1));
724  } else {
725  total += (unique_path[i].pweight / zero_fraction) / ((unique_depth - i)
726  / static_cast<bst_float>(unique_depth + 1));
727  }
728  }
729  return total;
730 }

732 // recursive computation of SHAP values for a decision tree
733 inline void RegTree::TreeShap(const RegTree::FVec& feat, bst_float *phi,
734  unsigned node_index, unsigned unique_depth,
735  PathElement *parent_unique_path, bst_float parent_zero_fraction,
736  bst_float parent_one_fraction, int parent_feature_index,
737  int condition, unsigned condition_feature,
738  bst_float condition_fraction) const {
739  const auto node = (*this)[node_index];
740 
741  // stop if we have no weight coming down to us
742  if (condition_fraction == 0) return;
743 
744  // extend the unique path
745  PathElement *unique_path = parent_unique_path + unique_depth + 1;
746  std::copy(parent_unique_path, parent_unique_path + unique_depth + 1, unique_path);
747 
748  if (condition == 0 || condition_feature != static_cast<unsigned>(parent_feature_index)) {
749  ExtendPath(unique_path, unique_depth, parent_zero_fraction,
750  parent_one_fraction, parent_feature_index);
751  }
752  const unsigned split_index = node.split_index();
753 
754  // leaf node
755  if (node.is_leaf()) {
756  for (unsigned i = 1; i <= unique_depth; ++i) {
757  const bst_float w = UnwoundPathSum(unique_path, unique_depth, i);
758  const PathElement &el = unique_path[i];
759  phi[el.feature_index] += w * (el.one_fraction - el.zero_fraction)
760  * node.leaf_value() * condition_fraction;
761  }
762 
763  // internal node
764  } else {
765  // find which branch is "hot" (meaning x would follow it)
766  unsigned hot_index = 0;
767  if (feat.is_missing(split_index)) {
768  hot_index = node.cdefault();
769  } else if (feat.fvalue(split_index) < node.split_cond()) {
770  hot_index = node.cleft();
771  } else {
772  hot_index = node.cright();
773  }
774  const unsigned cold_index = (static_cast<int>(hot_index) == node.cleft() ?
775  node.cright() : node.cleft());
776  const bst_float w = this->stat(node_index).sum_hess;
777  const bst_float hot_zero_fraction = this->stat(hot_index).sum_hess / w;
778  const bst_float cold_zero_fraction = this->stat(cold_index).sum_hess / w;
779  bst_float incoming_zero_fraction = 1;
780  bst_float incoming_one_fraction = 1;
781 
782  // see if we have already split on this feature,
783  // if so we undo that split so we can redo it for this node
784  unsigned path_index = 0;
785  for (; path_index <= unique_depth; ++path_index) {
786  if (static_cast<unsigned>(unique_path[path_index].feature_index) == split_index) break;
787  }
788  if (path_index != unique_depth + 1) {
789  incoming_zero_fraction = unique_path[path_index].zero_fraction;
790  incoming_one_fraction = unique_path[path_index].one_fraction;
791  UnwindPath(unique_path, unique_depth, path_index);
792  unique_depth -= 1;
793  }
794 
795  // divide up the condition_fraction among the recursive calls
796  bst_float hot_condition_fraction = condition_fraction;
797  bst_float cold_condition_fraction = condition_fraction;
798  if (condition > 0 && split_index == condition_feature) {
799  cold_condition_fraction = 0;
800  unique_depth -= 1;
801  } else if (condition < 0 && split_index == condition_feature) {
802  hot_condition_fraction *= hot_zero_fraction;
803  cold_condition_fraction *= cold_zero_fraction;
804  unique_depth -= 1;
805  }
806 
807  TreeShap(feat, phi, hot_index, unique_depth + 1, unique_path,
808  hot_zero_fraction * incoming_zero_fraction, incoming_one_fraction,
809  split_index, condition, condition_feature, hot_condition_fraction);
810 
811  TreeShap(feat, phi, cold_index, unique_depth + 1, unique_path,
812  cold_zero_fraction * incoming_zero_fraction, 0,
813  split_index, condition, condition_feature, cold_condition_fraction);
814  }
815 }

817 inline void RegTree::CalculateContributions(const RegTree::FVec& feat, unsigned root_id,
818  bst_float *out_contribs,
819  int condition,
820  unsigned condition_feature) const {
821  // find the expected value of the tree's predictions
822  if (condition == 0) {
823  bst_float node_value = this->node_mean_values[static_cast<int>(root_id)];
824  out_contribs[feat.size()] += node_value;
825  }
826 
827  // Preallocate space for the unique path data
828  const int maxd = this->MaxDepth(root_id) + 2;
829  PathElement *unique_path_data = new PathElement[(maxd * (maxd + 1)) / 2];
830 
831  TreeShap(feat, out_contribs, root_id, 0, unique_path_data,
832  1, 1, -1, condition, condition_feature, 1);
833  delete[] unique_path_data;
834 }
