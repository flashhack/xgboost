29 inline double CoordinateDelta(double sum_grad, double sum_hess, double w,
30  double reg_lambda, double reg_alpha,
31  double sum_instance_weight) {
32  reg_alpha *= sum_instance_weight;
33  reg_lambda *= sum_instance_weight;
34  if (sum_hess < 1e-5f) return 0.0f;
35  double tmp = w - (sum_grad + reg_lambda * w) / (sum_hess + reg_lambda);
36  if (tmp >= 0) {
37  return std::max(
38  -(sum_grad + reg_lambda * w + reg_alpha) / (sum_hess + reg_lambda), -w);
39  } else {
40  return std::min(
41  -(sum_grad + reg_lambda * w - reg_alpha) / (sum_hess + reg_lambda), -w);
42  }
43 }

70 inline std::pair<double, double> GetGradient(
71  int group_idx, int num_group, int fidx, const std::vector<bst_gpair> &gpair,
72  DMatrix *p_fmat) {
73  double sum_grad = 0.0, sum_hess = 0.0;
74  dmlc::DataIter<ColBatch> *iter = p_fmat->ColIterator();
75  while (iter->Next()) {
76  const ColBatch &batch = iter->Value();
77  ColBatch::Inst col = batch[fidx];
78  const bst_omp_uint ndata = static_cast<bst_omp_uint>(col.length);
79  for (bst_omp_uint j = 0; j < ndata; ++j) {
80  const bst_float v = col[j].fvalue;
81  auto &p = gpair[col[j].index * num_group + group_idx];
82  if (p.GetHess() < 0.0f) continue;
83  sum_grad += p.GetGrad() * v;
84  sum_hess += p.GetHess() * v * v;
85  }
86  }

281 class GreedyFeatureSelector : public FeatureSelector {
282  public:
283  int SelectNextFeature(int iteration, const gbm::GBLinearModel &model,
284  int group_idx, const std::vector<bst_gpair> &gpair,
285  DMatrix *p_fmat, float alpha, float lambda,
286  double sum_instance_weight) override {
287  // Find best
288  int best_fidx = 0;
289  double best_weight_update = 0.0f;
290 
291  for (auto fidx = 0U; fidx < model.param.num_feature; fidx++) {
292  const float w = model[fidx][group_idx];
293  auto gradient = GetGradientParallel(
294  group_idx, model.param.num_output_group, fidx, gpair, p_fmat);
295  float dw = static_cast<float>(
296  CoordinateDelta(gradient.first, gradient.second, w, lambda, alpha,
297  sum_instance_weight));
298  if (std::abs(dw) > std::abs(best_weight_update)) {
299  best_weight_update = dw;
300  best_fidx = fidx;
301  }
302  }
